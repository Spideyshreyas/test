import numpy as np
a=np.array([10,15,20,15,30])
print(a)
sum=np.sum(a)
print(sum)
avg=np.mean(a)
print(avg)
med=np.median(a)
print(med)
min=np.min(a)
print(min)
max=np.max(a)
print(max)
minind=np.argmin(a)
print(minind)
maxind=np.argmax(a)
print(maxind)
stdev=np.std(a)
print(stdev)
vari=np.var(a)
print(vari)
